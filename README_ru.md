
## Класс Date
[тег]: Date
```java
// Класс для работы с датой
class Date

    // Конструктор
    Date()

     // Вернуть год
     //   возвращает: Целое число
    int getYear()

    // Установить год
    //   параметр year: Целое число
    void setYear(int year)

     // Вернуть месяц
     //   возвращает: Целое число от 1 до 12 включительно
    int getMonth()

    // Устаовить месяц
    //   параметр month: Целое число от 1 до 12 включительно
    void setMonth(int month)

    // Вернуть день
    //   возвращает: Целое число от 1 до 28,29,30,31 включительно
    //               в зависости от месяца и года
    int getDay()

    // Установить день
    //   параметр day: Целое число от 1 до 28,29,30,31 включительно
    //                 в зависости от месяца и года
    void setDay(int day)

    // Вернуть часы
    //   возвращает: Целое число от 0 до 23 включительно
    int getHours()

    // Установить часы
    //   параметр hours: Целое число от 0 до 23 включительно
    void setHours(int hours)

    // Вернуть минуты
    //   возвращает: Целое число от 0 до 59 включительно
    int getMinutes()

    // Установить минуты
    //   параметр minutes: Целое число от 0 до 59 включительно
    void setMinutes(int minutes)

    // Вернуть секунды
    //   возвращает: Целое число от 0 до 59 включительно
    int getSeconds()

    // Установить секунды
    //   параметр seconds: Целое число от 0 до 59 включительно
    void setSeconds(int seconds)
```

## Класс WebDriver
[тег]: WebDriver
```java
// Класс для работы с веб браузером
class WebDriver 

    // Конструктор
    //   параметр type Тип браузера("Chrome","Firefox","Htmlunit","Ie")
    WebDriver(String type)

    // Открыть страницу сайта в браузере
    //   параметр url Адрес страницы сайта
    void open(String url)
    
    // Закрыть браузер
    void quit()

    // Вернуть адрес страницы
    //   возвращает: Адрес страницы
    String getUrl()

    // Вернуть заголовок страницы
    //   возвращает: Заголовок страницы
    String getTitle()

    // Сделать и сохранить скриншот экрана
    //   параметр screenshotSavePath Путь куда сохранить скриншот
    void getScreenshot(String screenshotSavePath)

    // Вернуть положение окна браузера по X относительно верхнего левого угла
    //   возвращает: Координата X окна браузера
    int getX()

    // Установить положение окна браузера по X относительно верхнего левого угла
    //   параметр x Координата X окна браузера
    void setX(int x) 

    // Вернуть положение окна браузера по Y относительно верхнего левого угла
    //   возвращает: Координата Y окна браузера
    int getY()
    
    // Установить положение окна браузера по Y относительно верхнего левого угла
    //   параметр y Координата Y окна браузера
    void setY(int y)
    
    // Вернуть ширину окна браузера
    //   возвращает: Ширина окна браузера
    int getWidth()

    // Установить ширину окна браузера
    //   параметр width Ширина окна браузера
    void setWidth(int width)

    // Вернуть высоту окна браузера
    //   возвращает: Высота окна браузера
    int getHeight()

    // Установить высоту окна браузера
    //   параметр height Высота окна браузера
    void setHeight(int height) 

    // Максимализировать("распахнуть") окно браузера во весь экран
    void maximize()

    // Найти элемент по CSS пути
    //   параметр cssPath CSS путь(селектор) элемента
    //   возвращает: Веб элемент или null если не найден
    WebElement findElement(String cssPath)

    // Найти элементы по CSS пути
    //   параметр cssPath CSS путь(селектор) элементов
    //   возвращает: Список веб элементов
    List<WebElement> findElements(String cssPath)

    // Вернуть список Cookie файлов
    //   возвращает: Сисок Cookie файлов 
    List<Cookie> getCookies()

    // Добавить Cookie файл
    //   параметр cookie Cookie файл для добавления
    void addCoockie(Cookie cookie)
    
    // Удалить Cookie файл
    //   параметр cookie Cookie файл для удаления 
    void deleteCookie(Cookie cookie)

    // Удалить все Cookie файлы
    void deleteAllCookies()
    
    // Выполнить JavaScript строку
    //   параметр script Строка JavaScript
    //   параметр args Необязательные аргументы в JavaScript(могут быть получены в
    //              JavaScript через arguments[<номер аргумента>]
    void executeJavaScript(String script, Object ...args)


/* Пример работы с классом WebDriver

*/
```


## Класс Actions
[тег]: Actions
```java
// Класс для
class Actions

    // Конструктор
    //   параметр webDriver Веб драйвер
    Actions(WebDriver webDriver)

    // Кликнуть левой кнопкой мыши
    //   возвращает: Объект Список действий
    Actions click()

    // Кликнуть левой кнопкой мыши по веб элементу
    //   параметр webElement Веб элемент
    //   возвращает: Объект Список действий
    Actions click(WebElement webElement)

    // Кликнуть по содержимому
    //   возвращает: Объект Список действий
    Actions clickContext()
    
    // Кликнуть по содержимому веб элемента
    //   параметр webElement Веб элемент
    //   возвращает: Объект Список действий
    Actions clickContext(WebElement webElement)
    
    // Зажать левую кнопку мыши
    //   возвращает: Объект Список действий
    Actions clickAndHold()

    // Зажать левую кнопку мыши на веб элементе
    //   параметр webElement Веб элемент
    //   возвращает: Объект Список действий
    Actions clickAndHold(WebElement webElement) 

    // Отпустить левую кнопку мыши
    //   возвращает: Объект Список действий
    Actions releaseHold()

    // Отпустить левую кнопку мыши на веб элементе
    //   параметр webElement Веб элемент
    //   возвращает: Объект Список действий
    Actions releaseHold(WebElement webElement)

    // Кликнуть левой кнопкой мыши два раза
    //   возвращает: Объект Список действий
    Actions doubleClick()

    // Кликнуть левой кнопкой мыши два раза на веб элемент
    //   параметр webElement Веб элемент
    //   возвращает: Объект Список действий
    Actions doubleClick(WebElement webElement)

    // Нажать клавишу и удерживать её
    //   параметр key Код клавиши
    //   возвращает: Объект Список действий
    Actions keyDown(CharSequence key)

    // Нажать клавишу на веб элементе и удерживать её
    //   параметр webElement Веб элемент
    //   параметр key Код клавиши
    //   возвращает: Объект Список действий
    Actions keyDown(WebElement webElement, CharSequence key)

    // Отпустить нажатую клавишу
    //   параметр key Код клавиши
    //   возвращает: Объект Список действий
    Actions keyUp(CharSequence key)
    
    // Отпустить нажатую клавишу на веб элементе
    //   параметр webElement Веб элемент
    //   параметр key Код клавиши
    //   возвращает: Объект Список действий
    Actions keyUp(WebElement webElement, CharSequence key)

    // Отправить нажатие клавиш
    //   параметр keys Список клавиш
    //   возвращает: Объект Список действий
    Actions sendKeys(CharSequence ...keys)
    
    // Отправить нажатие клавиш веб элементу
    //   параметр element Веб элемент
    //   параметр keys Список кодов клавиш
    //   возвращает: Объект Список действий
    Actions sendKeys(WebElement webElement, CharSequence ...keys)

    // Сместить курсор мыши на заданные отспупы
    //   параметр xOffset Смещение по X
    //   параметр yOffset Смещение по Y
    //   возвращает: Объект Список действий
    Actions mouseMove(int xOffset, int yOffset)

    // Переместить курсор мыши на веб элемент
    //   параметр element Веб элемент
    //   возвращает: Объект Список действий
    Actions mouseMove(WebElement element)
    
    // Перенести один веб элемент на другой
    //   параметр source Веб элемент который будет перетаскиваться
    //   параметр target Веб элемент на который будет перетаскиваться веб
    //                элемент
    //   возвращает: Объект Список действий
    Actions dragAndDrop(WebElement source, WebElement target)

    // Перенести один веб элемент на заданное расстояние
    //   параметр webElement Веб элемент
    //   параметр xOffset Смещение по X
    //   параметр yOffset Смещение по Y
    //   возвращает: Объект список действий
    Actions dragAndDrop(WebElement webElement, int xOffset, int yOffset)

    // Выполнить действия
    void perform()


/* Пример работы с классом Actions

*/
```
