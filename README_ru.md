
## Класс Date
[тег]: Date
```java
// Класс для работы с датой
class Date

    // Конструктор
    Date()

     // Вернуть год
     //   возвращает: Целое число
    int getYear()

    // Установить год
    //   параметр year: Целое число
    void setYear(int year)

     // Вернуть месяц
     //   возвращает: Целое число от 1 до 12 включительно
    int getMonth()

    // Устаовить месяц
    //   параметр month: Целое число от 1 до 12 включительно
    void setMonth(int month)

    // Вернуть день
    //   возвращает: Целое число от 1 до 28,29,30,31 включительно
    //               в зависости от месяца и года
    int getDay()

    // Установить день
    //   параметр day: Целое число от 1 до 28,29,30,31 включительно
    //                 в зависости от месяца и года
    void setDay(int day)

    // Вернуть часы
    //   возвращает: Целое число от 0 до 23 включительно
    int getHours()

    // Установить часы
    //   параметр hours: Целое число от 0 до 23 включительно
    void setHours(int hours)

    // Вернуть минуты
    //   возвращает: Целое число от 0 до 59 включительно
    int getMinutes()

    // Установить минуты
    //   параметр minutes: Целое число от 0 до 59 включительно
    void setMinutes(int minutes)

    // Вернуть секунды
    //   возвращает: Целое число от 0 до 59 включительно
    int getSeconds()

    // Установить секунды
    //   параметр seconds: Целое число от 0 до 59 включительно
    void setSeconds(int seconds)


/* Пример работы с классом Date

*/
```


## Класс Cookie

```java
// Класс для работы с Cookie
class Cookie

    // Конструктор
    Cookie()

    // Вернуть домен
    //   возвращает: Строка с доменом
    String getDomain()

    // Установить домен
    //   параметр domain: Строка с доменом
    void setDomain(String domain)

    // Вернуть путь
    //   возвращает: Строка с путём
    String getPath()

    // Установить путь
    //   параметр path: Строка с путём
    void setPath(String path)

    // Вернуть имя
    //   возвращает: Строка с именем
    String getName()

    // Установить имя
    //   параметр name: Строка с именем
    void setName(String name)

    // Вернуть значение
    //   возвращает: Строка со значением
    String getValue()

    // Установить значение
    //   параметр value: Строка со значением
    void setValue(String value)

    // Вернуть дату "протухания" cookie
    //   возвращает: Объект класса Date (Дата)
    Date getExpiry()

    // !!! Может вернуть null если сервер не установил дату "протухания" cookie !!!
    // Установить дату "протухания" cookie
    //   параметр expiry: Объект класса Date (Дата)
    void setExpiry(Date expiry)


/* Пример работы с классом Cookie

*/

```


## Класс WebElement
```java
// Класс для работы с веб элементами страницы
class WebElement

    // Вернуть координату X вернего левого угла элемента
    //   возвращает: Целое число координату X верхнего левого угла веб элемента
    int getX()

    // Вернуть координату Y вернего левого угла элемента
    //   возвращает: Целое число координату Y верхнего левого угла веб элемента
    int getY()


    // Вернуть ширину элемента
    //   возвращает: Целое число ширина веб элемента
    int getWidth()


    // Вернуть высоту элемента
    //   возвращает: Целое число высота веб элемента
    int getHeight()


    // Элемент отображается на странице?
    //   возвращает: Булевское значение Отображается?
    boolean isDisplayed()


    // Элемент включён?
    //   возвращает: Булевское значение Включён?
    boolean isEnabled()

    // Элемент выбран?
    //   возвращает: Булевское значение Выбран?
    boolean isSelected()
```
