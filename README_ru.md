    int getYear()

    //   параметр day: число от 1 до 28,29,30,31 включительно

    void setDay(int day)

    void setHours(int hours)

    //   возвращает: число от 0 до 59 включительно
    int getMinutes()

    // Установить минуты
    //   параметр minutes: число от 0 до 59 включительно
    void setMinutes(int minutes)

    //   возвращает: число от 0 до 59 включительно
    int getSeconds()

    //   параметр seconds: число от 0 до 59 включительно
